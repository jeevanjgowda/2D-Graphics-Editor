#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

#define CANVAS_HEIGHT 20
#define CANVAS_WIDTH  60
#define MAX_SHAPES    100

typedef enum {
    SHAPE_LINE, SHAPE_RECTANGLE, SHAPE_CIRCLE, SHAPE_TRIANGLE
} ShapeType;

typedef struct { int x1, y1, x2, y2; } LineParams;
typedef struct { int x1, y1, x2, y2; } RectParams;
typedef struct { int cx, cy, radius;  } CircleParams;
typedef struct { int x1,y1,x2,y2,x3,y3; } TriangleParams;

typedef struct {
    int id;
    ShapeType type;
    union {
        LineParams     line;
        RectParams     rect;
        CircleParams   circle;
        TriangleParams triangle;
    } params;
} Shape;

char  canvas[CANVAS_HEIGHT][CANVAS_WIDTH];
Shape shapes[MAX_SHAPES];
int   shape_count  = 0;
int   next_shape_id = 1;

void clear_canvas()  { /* fill canvas[][] with '_' */ }
void display_canvas(){ /* print column headers + bordered rows */ }
void set_pixel(int x, int y) {
    if (x>=0 && x<CANVAS_WIDTH && y>=0 && y<CANVAS_HEIGHT)
        canvas[y][x] = '*';
}

/* Bresenham's Line Algorithm */
void draw_line(int x1,int y1,int x2,int y2) {
    int dx=abs(x2-x1), dy=abs(y2-y1);
    int sx=(x1<x2)?1:-1, sy=(y1<y2)?1:-1;
    int err=dx-dy;
    while(1){
        set_pixel(x1,y1);
        if(x1==x2 && y1==y2) break;
        int e2=2*err;
        if(e2>-dy){ err-=dy; x1+=sx; }
        if(e2< dx){ err+=dx; y1+=sy; }
    }
}

/* Rectangle border */
void draw_rect(int x1,int y1,int x2,int y2) {
    if(x1>x2){int t=x1;x1=x2;x2=t;}
    if(y1>y2){int t=y1;y1=y2;y2=t;}
    for(int x=x1;x<=x2;x++){ set_pixel(x,y1); set_pixel(x,y2); }
    for(int y=y1;y<=y2;y++){ set_pixel(x1,y); set_pixel(x2,y); }
}

/* Midpoint Circle Algorithm - 8-point symmetry */
static void draw_circle_points(int xc,int yc,int x,int y){
    set_pixel(xc+x,yc+y); set_pixel(xc-x,yc+y);
    set_pixel(xc+x,yc-y); set_pixel(xc-x,yc-y);
    set_pixel(xc+y,yc+x); set_pixel(xc-y,yc+x);
    set_pixel(xc+y,yc-x); set_pixel(xc-y,yc-x);
}
void draw_circle(int xc,int yc,int r){
    int x=0, y=r, d=3-2*r;
    draw_circle_points(xc,yc,x,y);
    while(y>=x){
        x++;
        if(d>0){ y--; d=d+4*(x-y)+10; } else { d=d+4*x+6; }
        draw_circle_points(xc,yc,x,y);
    }
}

/* Triangle via 3 lines */
void draw_triangle(int x1,int y1,int x2,int y2,int x3,int y3){
    draw_line(x1,y1,x2,y2);
    draw_line(x2,y2,x3,y3);
    draw_line(x3,y3,x1,y1);
}

/* Re-render all shapes from scratch */
void render_all_shapes(){
    clear_canvas();
    for(int i=0;i<shape_count;i++){
        Shape s=shapes[i];
        switch(s.type){
            case SHAPE_LINE:
                draw_line(s.params.line.x1,s.params.line.y1,
                          s.params.line.x2,s.params.line.y2); break;
            case SHAPE_RECTANGLE:
                draw_rect(s.params.rect.x1,s.params.rect.y1,
                          s.params.rect.x2,s.params.rect.y2); break;
            case SHAPE_CIRCLE:
                draw_circle(s.params.circle.cx,s.params.circle.cy,
                            s.params.circle.radius); break;
            case SHAPE_TRIANGLE:
                draw_triangle(s.params.triangle.x1,s.params.triangle.y1,
                              s.params.triangle.x2,s.params.triangle.y2,
                              s.params.triangle.x3,s.params.triangle.y3); break;
        }
    }
}

/* Robust integer input using fgets + strtol */
int read_int(const char *prompt,int min_val,int max_val){
    int val; char buffer[100];
    while(1){
        printf("%s",prompt);
        if(!fgets(buffer,sizeof(buffer),stdin)) continue;
        buffer[strcspn(buffer,"\n")]='\0';
        char *ep; val=(int)strtol(buffer,&ep,10);
        if(ep==buffer||*ep!='\0'){
            printf("Invalid input. Enter a valid integer.\n"); continue;
        }
        if(val<min_val||val>max_val){
            printf("Out of range (%d-%d).\n",min_val,max_val); continue;
        }
        return val;
    }
}

/* Cross-platform screen clear */
void clear_screen(){
#ifdef _WIN32
    system("cls");
#else
    printf("\033[H\033[J");
#endif
}

/* Main menu loop, add_shape_menu(), delete_shape_menu(),
   modify_shape_menu(), list_shapes() — all implemented
   with switch-case blocks for each shape type */

int main(){
    clear_canvas();
    while(1){
        clear_screen();
        printf("=== 2D GRAPHICS EDITOR ===\n\n");
        render_all_shapes();
        display_canvas();
        list_shapes();
        printf("1.Add  2.Delete  3.Modify  4.Clear  5.Exit\n");
        int choice=read_int("Choice (1-5): ",1,5);
        switch(choice){
            case 1: add_shape_menu();    break;
            case 2: delete_shape_menu(); break;
            case 3: modify_shape_menu(); break;
            case 4: shape_count=0; break;
            case 5: printf("Goodbye!\n"); return 0;
        }
    }
}

void display_canvas() {
    /* Line 1: tens digit of column index */
    printf("   ");
    for (int x = 0; x < CANVAS_WIDTH; x++) {
        if (x % 10 == 0) printf("%d", x / 10);
        else printf(" ");
    }
    printf("\n   ");

    /* Line 2: units digit of column index */
    for (int x = 0; x < CANVAS_WIDTH; x++)
        printf("%d", x % 10);
    printf("\n");

    /* Top border */
    printf("  +");
    for (int x = 0; x < CANVAS_WIDTH; x++) putchar('-');
    printf("+\n");

    /* Canvas rows with row number on left */
    for (int y = 0; y < CANVAS_HEIGHT; y++) {
        printf("%2d|", y);
        for (int x = 0; x < CANVAS_WIDTH; x++)
            putchar(canvas[y][x]);
        printf("|\n");
    }

    /* Bottom border */
    printf("  +");
    for (int x = 0; x < CANVAS_WIDTH; x++) putchar('-');
    printf("+\n");
}

The canvas is initialized to '_' (underscores) so empty cells
are visible. Drawn pixels become '*'. The bordered display
with coordinate headers makes it easy to enter coordinates.

Each shape stores an int id field. next_shape_id increments
on every add. To delete or modify, search the shapes[] array
for a matching ID.

Delete (shifts remaining shapes left):
void delete_shape_menu() {
    list_shapes();
    int del_id = read_int("Enter ID to delete (0=cancel): ",
                           0, next_shape_id-1);
    if (del_id == 0) return;
    int idx = -1;
    for (int i = 0; i < shape_count; i++)
        if (shapes[i].id == del_id) { idx = i; break; }
    if (idx == -1) { printf("ID not found!\n"); return; }
    /* Shift left */
    for (int i = idx; i < shape_count-1; i++)
        shapes[i] = shapes[i+1];
    shape_count--;
    printf("Shape %d deleted.\n", del_id);
}

Modify (get pointer to shape, re-enter params):
void modify_shape_menu() {
    list_shapes();
    int mod_id = read_int("Enter ID to modify (0=cancel): ",
                           0, next_shape_id-1);
    if (mod_id == 0) return;
    Shape *s = NULL;
    for (int i = 0; i < shape_count; i++)
        if (shapes[i].id == mod_id) { s = &shapes[i]; break; }
    if (!s) { printf("ID not found!\n"); return; }
    /* Re-enter coordinates based on s->type (switch-case) */
    /* Same input prompts as add_shape_menu() */
    printf("Shape %d modified.\n", mod_id);
}

After delete or modify, render_all_shapes() is called at the
start of the next loop iteration, so the canvas auto-updates.
